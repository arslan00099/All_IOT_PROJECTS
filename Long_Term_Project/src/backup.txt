void resetSPIFFS()
{
  // Unmount the SPIFFS file system
  SPIFFS.end();

  // Format the SPIFFS file system
  SPIFFS.format();

  // Mount the SPIFFS file system
  if (!SPIFFS.begin())
  {
    Serial.println("Failed to mount file system");
    return;
  }
}
